27pub fn start_comments(w: &mut dyn Write, conf: &Config) -> Result<()> {
28    if conf.single_version_file.is_some() {
29        start_comments_no_version(w, conf)
30    } else {
31        single_version_file(w, conf, "// ")?;
32        writeln!(w, "// DO NOT EDIT")
33    }
34}

36pub fn start_comments_no_version(w: &mut dyn Write, conf: &Config) -> Result<()> {
37    writeln!(
38        w,
39        "// This file was generated by gir (https://github.com/gtk-rs/gir)"
40    )?;
41    write!(
42        w,
43        "{}",
44        conf.girs_version
45            .iter()
46            .fold(String::new(), |mut output, info| {
47                writeln!(
48                    output,
49                    "// from {}{}",
50                    info.gir_dir.display(),
51                    info.get_repository_url()
52                        .map_or_else(String::new, |url| format!(" ({url})")),
53                )
54                .unwrap();
55                output
56            })
57    )?;
58    writeln!(w, "// DO NOT EDIT")?;
59    Ok(())

62pub fn single_version_file(w: &mut dyn Write, conf: &Config, prefix: &str) -> Result<()> {
63    write!(
64        w,
65        "{}Generated by gir (https://github.com/gtk-rs/gir @ {})
66{}",
67        prefix,
68        VERSION,
69        conf.girs_version
70            .iter()
71            .fold(String::new(), |mut output, info| {
72                match (info.get_repository_url(), info.get_hash()) {
73                    (Some(url), Some(hash)) => writeln!(
74                        output,
75                        "{}from {} ({} @ {})",
76                        prefix,
77                        info.gir_dir.display(),
78                        url,
79                        hash,
80                    ),
81                    (None, Some(hash)) => {
82                        writeln!(
83                            output,
84                            "{}from {} (@ {})",
85                            prefix,
86                            info.gir_dir.display(),
87                            hash,
88                        )
89                    }
90                    _ => writeln!(output, "{}from {}", prefix, info.gir_dir.display()),
91                }
92                .unwrap();
93                output
94            }),
95    )
96}

98pub fn uses(
99    w: &mut dyn Write,
100    env: &Env,
101    imports: &Imports,
102    outer_version: Option<Version>,
103) -> Result<()> {
104    writeln!(w)?;
105
106    let mut grouped_imports: BTreeMap<(&str, Option<ImportConditions>), Vec<&str>> =
107        BTreeMap::new();
108
109    for (name, scope) in imports.iter() {
110        let (crate_name, import) = name.split_once("::").unwrap();
111        let mut scope = scope.clone();
112
113        // The check here is needed to group unneeded version guards and allow grouping those imports
114        scope.version = Version::if_stricter_than(scope.version, outer_version);
115        let to_compare_with = env.config.min_required_version(env, None);
116        scope.version = match (scope.version, to_compare_with) {
117            (Some(v), Some(to_compare_v)) => {
118                if v > to_compare_v {
119                    scope.version
120                } else {
121                    None
122                }
123            }
124            (Some(_), _) => scope.version,
125            _ => None,
126        };
127
128        let key = if scope.constraints.is_empty() && scope.version.is_none() {
129            (crate_name, None)
130        } else {
131            (crate_name, Some(scope))
132        };
133        grouped_imports
134            .entry(key)
135            .and_modify(|entry| entry.push(import))
136            .or_insert_with(|| vec![import]);
137    }
138
139    for ((crate_name, scope), names) in grouped_imports.iter() {
140        if !scope.is_none() {
141            let scope = scope.as_ref().unwrap();
142            if !scope.constraints.is_empty() {
143                if scope.constraints.len() == 1 {
144                    writeln!(w, "#[cfg({})]", scope.constraints[0])?;
145                } else {
146                    writeln!(w, "#[cfg(any({}))]", scope.constraints.join(", "))?;
147                }
148                writeln!(
149                    w,
150                    "#[cfg_attr(docsrs, doc(cfg({})))]",
151                    scope.constraints.join(", ")
152                )?;
153            }
154            version_condition(w, env, None, scope.version, false, 0)?;
155        }
156        writeln!(w, "use {crate_name}::{{{}}};", names.join(","))?;
157    }
158
159    Ok(())
160}

162fn format_parent_name(env: &Env, p: &StatusedTypeId) -> String {
163    if p.type_id.ns_id == namespaces::MAIN {
164        p.name.clone()
165    } else {
166        format!(
167            "{krate}::{name}",
168            krate = env.namespaces[p.type_id.ns_id].crate_name,
169            name = p.name,
170        )
171    }

174pub fn define_fundamental_type(
175    w: &mut dyn Write,
176    env: &Env,
177    type_name: &str,
178    glib_name: &str,
179    glib_func_name: &str,
180    ref_func: Option<&str>,
181    unref_func: Option<&str>,
182    parents: &[StatusedTypeId],
183    visibility: Visibility,
184    type_id: TypeId,
185) -> Result<()> {
186    let sys_crate_name = env.sys_crate_import(type_id);
187    writeln!(w, "{} {{", use_glib_type(env, "wrapper!"))?;
188    doc_alias(w, glib_name, "", 1)?;
189    external_doc_link(
190        w,
191        env.config.external_docs_url.as_deref(),
192        type_name,
193        &visibility,
194        1,
195    )?;
196    writeln!(
197        w,
198        "\t{visibility} struct {type_name}(Shared<{sys_crate_name}::{glib_name}>);"
199    )?;
200    writeln!(w)?;
201
202    writeln!(w, "\tmatch fn {{")?;
203    let (ref_fn, unref_fn, ptr, ffi_crate_name) = if parents.is_empty() {
204        // it's the super-type, it must have a ref/unref functions
205        (
206            ref_func.unwrap().to_owned(),
207            unref_func.unwrap().to_owned(),
208            "ptr".to_owned(),
209            sys_crate_name.to_owned(),
210        )
211    } else {
212        let (ref_fn, unref_fn, ptr, ffi_crate_name) = parents
213            .iter()
214            .find_map(|p| {
215                use crate::library::*;
216                let type_ = env.library.type_(p.type_id);
217                let parent_sys_crate_name = env.sys_crate_import(p.type_id);
218                match type_ {
219                    Type::Class(class) => Some((
220                        class.ref_fn.as_ref().unwrap().clone(),
221                        class.unref_fn.as_ref().unwrap().clone(),
222                        format!(
223                            "ptr as *mut {}::{}",
224                            parent_sys_crate_name,
225                            class.c_type.clone()
226                        ),
227                        parent_sys_crate_name,
228                    )),
229                    _ => None,
230                }
231            })
232            .unwrap();
233        // otherwise get that information from the parent
234        (ref_fn, unref_fn, ptr, ffi_crate_name)
235    };
236
237    writeln!(w, "\t\tref => |ptr| {ffi_crate_name}::{ref_fn}({ptr}),")?;
238    writeln!(w, "\t\tunref => |ptr| {ffi_crate_name}::{unref_fn}({ptr}),")?;
239
240    writeln!(w, "\t}}")?;
241    writeln!(w, "}}")?;
242
243    // We can't use type_ from glib::wrapper! because that would auto-implement
244    // Value traits which are often not the correct types
245    writeln!(w, "\n")?;
246    writeln!(w, "impl StaticType for {type_name} {{")?;
247    writeln!(w, "\tfn static_type() -> {} {{", use_glib_type(env, "Type"))?;
248    writeln!(
249        w,
250        "\t\t unsafe {{ from_glib({sys_crate_name}::{glib_func_name}()) }}"
251    )?;
252    writeln!(w, "\t}}")?;
253    writeln!(w, "}}")?;
254    Ok(())
255}

257pub fn define_object_type(
258    w: &mut dyn Write,
259    env: &Env,
260    type_name: &str,
261    glib_name: &str,
262    glib_class_name: Option<&str>,
263    glib_func_name: &str,
264    is_interface: bool,
265    parents: &[StatusedTypeId],
266    visibility: Visibility,
267    type_id: TypeId,
268) -> Result<()> {
269    let sys_crate_name = env.sys_crate_import(type_id);
270    let class_name = {
271        if let Some(s) = glib_class_name {
272            format!(", {sys_crate_name}::{s}")
273        } else {
274            String::new()
275        }
276    };
277
278    let kind_name = if is_interface { "Interface" } else { "Object" };
279
280    let parents: Vec<StatusedTypeId> = parents
281        .iter()
282        .filter(|p| !p.status.ignored())
283        .cloned()
284        .collect();
285
286    writeln!(w, "{} {{", use_glib_type(env, "wrapper!"))?;
287    doc_alias(w, glib_name, "", 1)?;
288    external_doc_link(
289        w,
290        env.config.external_docs_url.as_deref(),
291        type_name,
292        &visibility,
293        1,
294    )?;
295    if parents.is_empty() {
296        writeln!(
297            w,
298            "\t{visibility} struct {type_name}({kind_name}<{sys_crate_name}::{glib_name}{class_name}>);"
299        )?;
300    } else if is_interface {
301        let prerequisites: Vec<String> =
302            parents.iter().map(|p| format_parent_name(env, p)).collect();
303
304        writeln!(
305            w,
306            "\t{} struct {}(Interface<{}::{}{}>) @requires {};",
307            visibility,
308            type_name,
309            sys_crate_name,
310            glib_name,
311            class_name,
312            prerequisites.join(", ")
313        )?;
314    } else {
315        let interfaces: Vec<String> = parents
316            .iter()
317            .filter(|p| {
318                use crate::library::*;
319
320                matches!(
321                    *env.library.type_(p.type_id),
322                    Type::Interface { .. } if !p.status.ignored()
323                )
324            })
325            .map(|p| format_parent_name(env, p))
326            .collect();
327
328        let parents: Vec<String> = parents
329            .iter()
330            .filter(|p| {
331                use crate::library::*;
332
333                matches!(
334                    *env.library.type_(p.type_id),
335                    Type::Class { .. } if !p.status.ignored()
336                )
337            })
338            .map(|p| format_parent_name(env, p))
339            .collect();
340
341        let mut parents_string = String::new();
342        if !parents.is_empty() {
343            parents_string.push_str(format!(" @extends {}", parents.join(", ")).as_str());
344        }
345
346        if !interfaces.is_empty() {
347            if !parents.is_empty() {
348                parents_string.push(',');
349            }
350            parents_string.push_str(format!(" @implements {}", interfaces.join(", ")).as_str());
351        }
352
353        writeln!(
354            w,
355            "\t{visibility} struct {type_name}(Object<{sys_crate_name}::{glib_name}{class_name}>){parents_string};",
356        )?;
357    }
358    writeln!(w)?;
359    writeln!(w, "\tmatch fn {{")?;
360    writeln!(w, "\t\ttype_ => || {sys_crate_name}::{glib_func_name}(),")?;
361    writeln!(w, "\t}}")?;
362    writeln!(w, "}}")?;
363
364    Ok(())
365}

367fn define_boxed_type_internal(
368    w: &mut dyn Write,
369    env: &Env,
370    type_name: &str,
371    glib_name: &str,
372    copy_fn: &TraitInfo,
373    free_fn: &str,
374    boxed_inline: bool,
375    init_function_expression: &Option<String>,
376    copy_into_function_expression: &Option<String>,
377    clear_function_expression: &Option<String>,
378    get_type_fn: Option<&str>,
379    derive: &[Derive],
380    visibility: Visibility,
381    type_id: TypeId,
382) -> Result<()> {
383    let sys_crate_name = env.sys_crate_import(type_id);
384    writeln!(w, "{} {{", use_glib_type(env, "wrapper!"))?;
385
386    derives(w, derive, 1)?;
387    writeln!(
388        w,
389        "\t{} struct {}(Boxed{}<{}::{}>);",
390        visibility,
391        type_name,
392        if boxed_inline { "Inline" } else { "" },
393        sys_crate_name,
394        glib_name
395    )?;
396    writeln!(w)?;
397    writeln!(w, "\tmatch fn {{")?;
398    let mut_ov = if copy_fn.first_parameter_mut {
399        "mut_override(ptr)"
400    } else {
401        "ptr"
402    };
403    writeln!(
404        w,
405        "\t\tcopy => |ptr| {}::{}({}),",
406        sys_crate_name, copy_fn.glib_name, mut_ov
407    )?;
408    writeln!(w, "\t\tfree => |ptr| {sys_crate_name}::{free_fn}(ptr),")?;
409
410    if let (
411        Some(init_function_expression),
412        Some(copy_into_function_expression),
413        Some(clear_function_expression),
414    ) = (
415        init_function_expression,
416        copy_into_function_expression,
417        clear_function_expression,
418    ) {
419        writeln!(w, "\t\tinit => {init_function_expression},",)?;
420        writeln!(w, "\t\tcopy_into => {copy_into_function_expression},",)?;
421        writeln!(w, "\t\tclear => {clear_function_expression},",)?;
422    }
423
424    if let Some(get_type_fn) = get_type_fn {
425        writeln!(w, "\t\ttype_ => || {sys_crate_name}::{get_type_fn}(),")?;
426    }
427    writeln!(w, "\t}}")?;
428    writeln!(w, "}}")?;
429
430    Ok(())
431}

700pub fn cfg_deprecated_string(
701    env: &Env,
702    type_tid: Option<TypeId>,
703    deprecated: Option<Version>,
704    commented: bool,
705    indent: usize,
706) -> Option<String> {
707    let comment = if commented { "//" } else { "" };
708    deprecated.map(|v| {
709        if env.is_too_low_version(type_tid.map(|t| t.ns_id), Some(v)) {
710            format!("{}{}#[deprecated = \"Since {}\"]", tabs(indent), comment, v)
711        } else {
712            format!(
713                "{}{}#[cfg_attr({}, deprecated = \"Since {}\")]",
714                tabs(indent),
715                comment,
716                v.to_cfg(None),
717                v,
718            )
719        }
720    })
721}

723pub fn version_condition(
724    w: &mut dyn Write,
725    env: &Env,
726    ns_id: Option<u16>,
727    version: Option<Version>,
728    commented: bool,
729    indent: usize,
730) -> Result<()> {
731    if let Some(s) = version_condition_string(env, ns_id, version, commented, indent) {
732        writeln!(w, "{s}")?;
733    }
734    Ok(())

737pub fn version_condition_no_doc(
738    w: &mut dyn Write,
739    env: &Env,
740    ns_id: Option<u16>,
741    version: Option<Version>,
742    commented: bool,
743    indent: usize,
744) -> Result<()> {
745    let to_compare_with = env.config.min_required_version(env, ns_id);
746    let should_generate = match (version, to_compare_with) {
747        (Some(v), Some(to_compare_v)) => v > to_compare_v,
748        (Some(_), _) => true,
749        _ => false,
750    };
751    if should_generate {
752        // Prefix with the crate name if it's not the main one
753        let namespace_name = ns_id.and_then(|ns| {
754            if ns == namespaces::MAIN {
755                None
756            } else {
757                Some(env.namespaces.index(ns).crate_name.clone())
758            }
759        });
760        if let Some(s) = cfg_condition_string_no_doc(
761            Some(&version.unwrap().to_cfg(namespace_name.as_deref())),
762            commented,
763            indent,
764        ) {
765            writeln!(w, "{s}")?;
766        }
767    }
768    Ok(())
769}
770pub fn version_condition_doc(
771    w: &mut dyn Write,
772    env: &Env,
773    version: Option<Version>,
774    commented: bool,
775    indent: usize,
776) -> Result<()> {
777    match version {
778        Some(v) if v > env.config.min_cfg_version => {
779            if let Some(s) = cfg_condition_string_doc(Some(&v.to_cfg(None)), commented, indent) {
780                writeln!(w, "{s}")?;
781            }
782        }
783        _ => {}
784    }
785    Ok(())
786}

1025pub fn declare_default_from_new(
1026    w: &mut dyn Write,
1027    env: &Env,
1028    name: &str,
1029    functions: &[analysis::functions::Info],
1030    has_builder: bool,
1031) -> Result<()> {
1032    if let Some(func) = functions.iter().find(|f| {
1033        !f.hidden
1034            && f.status.need_generate()
1035            && f.name == "new"
1036            // Cannot generate Default implementation for Option<>
1037            && f.ret.parameter.as_ref().is_some_and(|x| !*x.lib_par.nullable)
1038            // Cannot generate Default implementation for Result<>
1039            && f.outs.mode != Mode::Throws(ThrowFunctionReturnStrategy::ReturnResult)
1040    }) {
1041        if func.parameters.rust_parameters.is_empty() {
1042            writeln!(w)?;
1043            version_condition(w, env, None, func.version, false, 0)?;
1044            writeln!(
1045                w,
1046                "impl Default for {name} {{
1047                     fn default() -> Self {{
1048                         Self::new()
1049                     }}
1050                 }}"
1051            )?;
1052        } else if has_builder {
1053            // create an alternative default implementation the uses `glib::object::Object::new()`
1054            writeln!(w)?;
1055            version_condition(w, env, None, func.version, false, 0)?;
1056            writeln!(
1057                w,
1058                "impl Default for {name} {{
1059                     fn default() -> Self {{
1060                         glib::object::Object::new::<Self>()
1061                     }}
1062                 }}"
1063            )?;
1064        }
1065    }
1066
1067    Ok(())
1068}

1070/// Escapes string in format suitable for placing inside double quotes.
1071pub fn escape_string(s: &str) -> String {
1072    let mut es = String::with_capacity(s.len() * 2);
1073    for c in s.chars() {
1074        match c {
1075            '\"' | '\\' => es.push('\\'),
1076            _ => (),
1077        }
1078        es.push(c);
1079    }
1080    es
1081}
