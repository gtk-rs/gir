21pub fn generate(w: &mut dyn Write, env: &Env, analysis: &analysis::object::Info) -> Result<()> {
22    general::start_comments(w, &env.config)?;
23    if analysis
24        .functions
25        .iter()
26        .any(|f| f.deprecated_version.is_some())
27    {
28        writeln!(w, "#![allow(deprecated)]")?;
29    }
30    general::uses(w, env, &analysis.imports, analysis.version)?;
31
32    let config = &env.config.objects[&analysis.full_name];
33    if config.default_value.is_some() {
34        log::error!(
35            "`default_value` can only be used on flags and enums. {} is neither. Ignoring \
36             `default_value`.",
37            analysis.name,
38        );
39    }
40
41    // Collect all supertypes that were added at a later time. The `glib::wrapper!`
42    // call needs to be done multiple times with different `#[cfg]` directives
43    // if there is a difference.
44    let mut ns_versions: BTreeMap<u16, BTreeMap<_, Vec<_>>> = BTreeMap::new();
45    for p in &analysis.supertypes {
46        use crate::library::*;
47
48        match *env.library.type_(p.type_id) {
49            Type::Interface(Interface { .. }) | Type::Class(Class { .. })
50                if !p.status.ignored() =>
51            {
52                let full_name = p.type_id.full_name(&env.library);
53                if let Some(object) = env.analysis.objects.get(&full_name) {
54                    let parent_version = object.version;
55                    let namespace_min_version = env
56                        .config
57                        .min_required_version(env, Some(object.type_id.ns_id));
58                    if parent_version > analysis.version && parent_version > namespace_min_version {
59                        ns_versions
60                            .entry(object.type_id.ns_id)
61                            .or_default()
62                            .entry(parent_version)
63                            .or_default()
64                            .push(p);
65                    }
66                }
67            }
68            _ => continue,
69        }
70    }
71    let namespaces: Vec<_> = ns_versions.into_iter().collect();
72
73    if namespaces.is_empty() || analysis.is_fundamental {
74        writeln!(w)?;
75        if analysis.is_fundamental {
76            general::define_fundamental_type(
77                w,
78                env,
79                &analysis.name,
80                &analysis.c_type,
81                &analysis.get_type,
82                analysis.ref_fn.as_deref(),
83                analysis.unref_fn.as_deref(),
84                &analysis.supertypes,
85                analysis.visibility,
86                analysis.type_id,
87            )?;
88        } else {
89            general::define_object_type(
90                w,
91                env,
92                &analysis.name,
93                &analysis.c_type,
94                analysis.c_class_type.as_deref(),
95                &analysis.get_type,
96                analysis.is_interface,
97                &analysis.supertypes,
98                analysis.visibility,
99                analysis.type_id,
100            )?;
101        }
102    } else {
103        // Write the `glib::wrapper!` calls from the highest version to the lowest.
104        // Each block is gated by its version feature and `not` of all higher version
105        // features. The base block is gated by `not(any(all versioned features))`.
106        let mut remove_types: HashSet<library::TypeId> = HashSet::new();
107
108        for (ns_id, versions) in &namespaces {
109            let namespace_name = if *ns_id == analysis::namespaces::MAIN {
110                None
111            } else {
112                Some(env.namespaces[*ns_id].crate_name.clone())
113            };
114
115            let all_version_cfgs: Vec<String> = versions
116                .keys()
117                .filter_map(|v| v.map(|v| v.to_cfg(namespace_name.as_deref())))
118                .collect();
119            let n = all_version_cfgs.len();
120            let mut rev_index = 0usize;
121
122            for (&version, stypes) in versions.iter().rev() {
123                let supertypes = analysis
124                    .supertypes
125                    .iter()
126                    .filter(|t| !remove_types.contains(&t.type_id))
127                    .cloned()
128                    .collect::<Vec<_>>();
129
130                writeln!(w)?;
131                if version.is_some() {
132                    let asc_index = n - 1 - rev_index;
133                    let positive = &all_version_cfgs[..=asc_index];
134                    let negative = all_version_cfgs.get(asc_index + 1);
135
136                    let mut parts: Vec<String> = positive.to_vec();
137                    if let Some(neg) = negative {
138                        parts.push(format!("not({neg})"));
139                    }
140
141                    if parts.len() == 1 {
142                        writeln!(w, "#[cfg({})]", parts[0])?;
143                    } else {
144                        writeln!(w, "#[cfg(all({}))]", parts.join(", "))?;
145                    }
146
147                    if rev_index == 0 {
148                        let highest_cfg = &all_version_cfgs[n - 1];
149                        writeln!(w, "#[cfg_attr(docsrs, doc(cfg({})))]", highest_cfg)?;
150                    }
151
152                    rev_index += 1;
153                }
154
155                general::define_object_type(
156                    w,
157                    env,
158                    &analysis.name,
159                    &analysis.c_type,
160                    analysis.c_class_type.as_deref(),
161                    &analysis.get_type,
162                    analysis.is_interface,
163                    &supertypes,
164                    analysis.visibility,
165                    analysis.type_id,
166                )?;
167
168                for t in stypes {
169                    remove_types.insert(t.type_id);
170                }
171            }
172
173            // Write the base `glib::wrapper!` gated by not(any(all versioned features)).
174            let supertypes = analysis
175                .supertypes
176                .iter()
177                .filter(|t| !remove_types.contains(&t.type_id))
178                .cloned()
179                .collect::<Vec<_>>();
180            writeln!(w)?;
181            if all_version_cfgs.len() == 1 {
182                writeln!(w, "#[cfg(not({}))]", all_version_cfgs[0])?;
183            } else {
184                let all = all_version_cfgs.join(", ");
185                writeln!(w, "#[cfg(not(any({all})))]")?;
186            }
187            general::define_object_type(
188                w,
189                env,
190                &analysis.name,
191                &analysis.c_type,
192                analysis.c_class_type.as_deref(),
193                &analysis.get_type,
194                analysis.is_interface,
195                &supertypes,
196                analysis.visibility,
197                analysis.type_id,
198            )?;
199        }
200    }
201
202    if (analysis.need_generate_inherent() && analysis.should_generate_impl_block())
203        || !analysis.final_type
204    {
205        writeln!(w)?;
206        write!(w, "impl {} {{", analysis.name)?;
207
208        if !analysis.final_type {
209            writeln!(
210                w,
211                "
212        pub const NONE: Option<&'static {}> = None;
213    ",
214                analysis.name
215            )?;
216        }
217
218        for func_analysis in &analysis.constructors() {
219            function::generate(
220                w,
221                env,
222                Some(analysis.type_id),
223                func_analysis,
224                Some(&analysis.specials),
225                analysis.version,
226                false,
227                false,
228                1,
229            )?;
230        }
231
232        if has_builder_properties(&analysis.builder_properties) {
233            // generate builder method that returns the corresponding builder
234            let builder_name = format!("{}Builder", analysis.name);
235            writeln!(
236                w,
237                "
238            // rustdoc-stripper-ignore-next
239            /// Creates a new builder-pattern struct instance to construct [`{name}`] objects.
240            ///
241            /// This method returns an instance of [`{builder_name}`](crate::builders::{builder_name}) which can be used to create [`{name}`] objects.
242            pub fn builder() -> {builder_name} {{
243                {builder_name}::new()
244            }}
245        ",
246                name = analysis.name,
247                builder_name = builder_name
248            )?;
249        }
250
251        if !analysis.need_generate_trait() {
252            for func_analysis in &analysis.methods() {
253                function::generate(
254                    w,
255                    env,
256                    Some(analysis.type_id),
257                    func_analysis,
258                    Some(&analysis.specials),
259                    analysis.version,
260                    false,
261                    false,
262                    1,
263                )?;
264            }
265
266            for property in &analysis.properties {
267                properties::generate(w, env, property, false, false, 1)?;
268            }
269
270            for child_property in &analysis.child_properties {
271                child_properties::generate(w, env, child_property, false, false, 1)?;
272            }
273        }
274
275        for func_analysis in &analysis.functions() {
276            function::generate(
277                w,
278                env,
279                Some(analysis.type_id),
280                func_analysis,
281                Some(&analysis.specials),
282                analysis.version,
283                false,
284                false,
285                1,
286            )?;
287        }
288
289        if !analysis.need_generate_trait() {
290            for signal_analysis in analysis
291                .signals
292                .iter()
293                .chain(analysis.notify_signals.iter())
294            {
295                signal::generate(w, env, signal_analysis, false, false, 1)?;
296            }
297        }
298
299        writeln!(w, "}}")?;
300
301        general::declare_default_from_new(
302            w,
303            env,
304            &analysis.name,
305            &analysis.functions,
306            has_builder_properties(&analysis.builder_properties),
307        )?;
308    }
309
310    trait_impls::generate(
311        w,
312        env,
313        &analysis.name,
314        &analysis.functions,
315        &analysis.specials,
316        if analysis.need_generate_trait() {
317            Some(&analysis.trait_name)
318        } else {
319            None
320        },
321        analysis.version,
322        None, // There is no need for #[cfg()] since it's applied on the whole file.
323    )?;
324
325    if has_builder_properties(&analysis.builder_properties) {
326        writeln!(w)?;
327        generate_builder(w, env, analysis)?;
328    }
329
330    if analysis.concurrency != library::Concurrency::None {
331        writeln!(w)?;
332    }
333
334    match analysis.concurrency {
335        library::Concurrency::Send | library::Concurrency::SendSync => {
336            writeln!(w, "unsafe impl Send for {} {{}}", analysis.name)?;
337        }
338        _ => (),
339    }
340
341    if let library::Concurrency::SendSync = analysis.concurrency {
342        writeln!(w, "unsafe impl Sync for {} {{}}", analysis.name)?;
343    }
344
345    if analysis.need_generate_trait() {
346        writeln!(w)?;
347        generate_trait(w, env, analysis)?;
348    }
349    Ok(())
350}

352fn generate_builder(w: &mut dyn Write, env: &Env, analysis: &analysis::object::Info) -> Result<()> {
353    let glib_crate_name = if env.namespaces.is_glib_crate {
354        "crate"
355    } else {
356        "glib"
357    };
358
359    writeln!(
360        w,
361        "// rustdoc-stripper-ignore-next
362        /// A [builder-pattern] type to construct [`{}`] objects.
363        ///
364        /// [builder-pattern]: https://doc.rust-lang.org/1.0.0/style/ownership/builders.html",
365        analysis.name,
366    )?;
367    writeln!(w, "#[must_use = \"The builder must be built to be used\"]")?;
368    writeln!(
369        w,
370        "pub struct {name}Builder {{
371            builder: {glib_name}::object::ObjectBuilder<'static, {name}>,
372        }}
373
374        impl {name}Builder {{
375        fn new() -> Self {{
376            Self {{ builder: {glib_name}::object::Object::builder() }}
377        }}",
378        name = analysis.name,
379        glib_name = glib_crate_name,
380    )?;
381    for (builder_props, super_tid) in &analysis.builder_properties {
382        for property in builder_props {
383            let direction = if property.is_get {
384                library::ParameterDirection::In
385            } else {
386                library::ParameterDirection::Out
387            };
388            let param_type = RustType::builder(env, property.typ)
389                .direction(direction)
390                .ref_mode(property.set_in_ref_mode)
391                .try_build();
392            let comment_prefix = if param_type.is_err() { "//" } else { "" };
393            let mut param_type_str = param_type.into_string();
394            let (param_type_override, bounds, conversion) = match param_type_str.as_str() {
395                "&str" => (
396                    Some(format!("impl Into<{glib_crate_name}::GString>")),
397                    String::new(),
398                    ".into()",
399                ),
400                "&[&str]" => (
401                    Some(format!("impl Into<{glib_crate_name}::StrV>")),
402                    String::from(""),
403                    ".into()",
404                ),
405                _ if !property.bounds.is_empty() => {
406                    let (bounds, _) = function::bounds(&property.bounds, &[], false, false);
407                    let param_bound = property.bounds.get_parameter_bound(&property.name);
408                    let alias = param_bound.map(|bound| {
409                        bound.full_type_parameter_reference(RefMode::ByRef, false, false)
410                    });
411                    let conversion = param_bound.and_then(|bound| match bound.bound_type {
412                        BoundType::AsRef(_) => Some(".as_ref().clone()"),
413                        _ => None,
414                    });
415                    (alias, bounds, conversion.unwrap_or(".clone().upcast()"))
416                }
417                typ if typ.starts_with('&') => {
418                    let should_clone =
419                        if let crate::library::Type::Record(record) = env.type_(property.typ) {
420                            match RecordType::of(record) {
421                                RecordType::Boxed => "",
422                                RecordType::AutoBoxed if !record.has_copy() => "",
423                                _ => ".clone()",
424                            }
425                        } else {
426                            ".clone()"
427                        };
428
429                    (None, String::new(), should_clone)
430                }
431                _ => (None, String::new(), ""),
432            };
433            if let Some(param_type_override) = param_type_override {
434                param_type_str = param_type_override.to_string();
435            }
436            let name = nameutil::mangle_keywords(nameutil::signal_to_snake(&property.name));
437
438            let version_condition_string =
439                version_condition_string(env, Some(super_tid.ns_id), property.version, false, 1);
440            let deprecated_string =
441                cfg_deprecated_string(env, Some(*super_tid), property.deprecated_version, false, 1);
442            let version_prefix = version_condition_string
443                .map(|version| format!("{comment_prefix}{version}\n"))
444                .unwrap_or_default();
445
446            let deprecation_prefix = deprecated_string
447                .map(|version| format!("{comment_prefix}{version}\n"))
448                .unwrap_or_default();
449
450            writeln!(
451                w,
452                "
453                        {version_prefix}{deprecation_prefix}    {comment_prefix}pub fn {name}{bounds}(self, {name}: {param_type_str}) -> Self {{
454                        {comment_prefix}    Self {{ builder: self.builder.property(\"{property_name}\", {name}{conversion}), }}
455                        {comment_prefix}}}",
456                property_name = property.name,
457            )?;
458        }
459    }
460
461    writeln!(
462        w,
463        "
464    // rustdoc-stripper-ignore-next
465    /// Build the [`{name}`].
466    #[must_use = \"Building the object from the builder is usually expensive and is not expected to have side effects\"]
467    pub fn build(self) -> {name} {{",
468        name = analysis.name,
469    )?;
470
471    // The split allows us to not have clippy::let_and_return lint disabled
472    if let Some(code) = analysis.builder_postprocess.as_ref() {
473        // We don't generate an assertion macro for the case where you have a build post-process
474        // as it is only used to initialize gtk in gtk::ApplicationBuilder which is too early to assert anything
475        writeln!(w, "    let ret = self.builder.build();")?;
476        writeln!(w, "        {{\n            {code}\n        }}")?;
477        writeln!(w, "    ret\n    }}")?;
478    } else {
479        if env.config.generate_safety_asserts {
480            writeln!(w, "{}", SafetyAssertionMode::InMainThread)?;
481        }
482        writeln!(w, "    self.builder.build() }}")?;
483    }
484    writeln!(w, "}}")
485}

487fn generate_trait(w: &mut dyn Write, env: &Env, analysis: &analysis::object::Info) -> Result<()> {
488    write!(
489        w,
490        "pub trait {}: IsA<{}> + 'static {{",
491        analysis.trait_name, analysis.name
492    )?;
493
494    for func_analysis in &analysis.methods() {
495        function::generate(
496            w,
497            env,
498            Some(analysis.type_id),
499            func_analysis,
500            Some(&analysis.specials),
501            analysis.version,
502            true,
503            false,
504            1,
505        )?;
506    }
507    for property in &analysis.properties {
508        properties::generate(w, env, property, true, false, 1)?;
509    }
510    for child_property in &analysis.child_properties {
511        child_properties::generate(w, env, child_property, true, false, 1)?;
512    }
513    for signal_analysis in analysis
514        .signals
515        .iter()
516        .chain(analysis.notify_signals.iter())
517    {
518        signal::generate(w, env, signal_analysis, true, false, 1)?;
519    }
520    writeln!(w, "}}")?;
521
522    writeln!(w)?;
523    writeln!(
524        w,
525        "impl<O: IsA<{}>> {} for O {{}}",
526        analysis.name, analysis.trait_name,
527    )?;
528
529    Ok(())
530}

532pub fn generate_reexports(
533    env: &Env,
534    analysis: &analysis::object::Info,
535    module_name: &str,
536    contents: &mut Vec<String>,
537    traits: &mut Vec<String>,
538    builders: &mut Vec<String>,
539) {
540    let mut cfgs: Vec<String> = Vec::new();
541    if let Some(cfg) = general::cfg_condition_string(analysis.cfg_condition.as_ref(), false, 0) {
542        cfgs.push(cfg);
543    }
544    if let Some(cfg) = general::version_condition_string(env, None, analysis.version, false, 0) {
545        cfgs.push(cfg);
546    }
547    if let Some(cfg) = general::cfg_deprecated_string(
548        env,
549        Some(analysis.type_id),
550        analysis.deprecated_version,
551        false,
552        0,
553    ) {
554        cfgs.push(cfg);
555    }
556
557    contents.push(String::new());
558    contents.extend_from_slice(&cfgs);
559    contents.push(format!("mod {module_name};"));
560    contents.extend_from_slice(&cfgs);
561
562    contents.push(format!(
563        "{} use self::{}::{};",
564        analysis.visibility.export_visibility(),
565        module_name,
566        analysis.name,
567    ));
568
569    if analysis.need_generate_trait() {
570        for cfg in &cfgs {
571            traits.push(format!("\t{cfg}"));
572        }
573        traits.push(format!(
574            "\tpub use super::{}::{};",
575            module_name, analysis.trait_name
576        ));
577    }
578
579    if has_builder_properties(&analysis.builder_properties) {
580        for cfg in &cfgs {
581            builders.push(format!("\t{cfg}"));
582        }
583        builders.push(format!(
584            "\tpub use super::{}::{}Builder;",
585            module_name, analysis.name
586        ));
587    }
