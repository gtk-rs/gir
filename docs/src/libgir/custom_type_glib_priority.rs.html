1//! Adds `glib::Priority` as custom type
2//! and attempts replace priority parameters with it in async functions
3
4use crate::{
5    analysis::conversion_type::ConversionType, config::WorkMode, library::*,
6    visitors::FunctionsMutVisitor,
7};
8
9impl Library {
10    pub fn add_glib_priority(&mut self, work_mode: WorkMode) {
11        if work_mode != WorkMode::Normal {
12            return;
13        }
14
15        let tid_int = self.find_type(0, "*.gint").expect("No basic type *.gint");
16        let glib_ns_id = self
17            .find_namespace("GLib")
18            .expect("Missing `GLib` namespace in add_glib_priority!");
19        let tid_priority = self.add_type(
20            glib_ns_id,
21            "Priority",
22            Type::Custom(Custom {
23                name: "Priority".to_string(),
24                conversion_type: ConversionType::Scalar,
25            }),
26        );
27
28        let mut replacer = ReplaceToPriority {
29            tid_priority,
30            tid_int,
31        };
32        self.namespace_mut(MAIN_NAMESPACE)
33            .visit_functions_mut(&mut replacer);
34    }
35}
36
37struct ReplaceToPriority {
38    pub tid_priority: TypeId,
39    pub tid_int: TypeId,
40}
41
42impl FunctionsMutVisitor for ReplaceToPriority {
43    fn visit_function_mut(&mut self, func: &mut Function) -> bool {
44        if !func.name.ends_with("_async") {
45            return true;
46        }
47        for par in &mut func.parameters {
48            if par.typ == self.tid_int && par.name.ends_with("priority") {
49                par.typ = self.tid_priority;
50            }
51        }
52        true
53    }
54}
